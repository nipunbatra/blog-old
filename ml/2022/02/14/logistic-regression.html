Generative model for logistic regression 

x = torch.linspace(-5, 15, 200)
x_dash = torch.vstack((torch.ones_like(x), x)).t()
theta = dist.MultivariateNormal(loc = torch.tensor([2., 0.]), covariance_matrix=torch.eye(2))
theta_sample = theta.sample()

p = torch.sigmoid(x_dash@theta_sample)

y = dist.Bernoulli(probs=p)
y_sample = y.sample()

plt.scatter(x, y_sample, c = y_sample)
