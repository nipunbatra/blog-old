def plot(f, x):

    mu =f(x).mean.reshape(-1, 1).flatten()
    var = np.diagonal(f(x).var.mat)
    std = np.sqrt(var)

    plt.plot(x, mu, color='purple', lw=4)
    plt.fill_between(x, mu-std, mu, alpha=0.1,color='purple')
    plt.fill_between(x, mu+std, mu, alpha=0.1, color='purple')

    plt.fill_between(x, mu-2*std, mu, alpha=0.1,color='purple')
    plt.fill_between(x, mu+2*std, mu, alpha=0.1, color='purple')
